JavaScript Control Example

// JavaScript control example const player = document.querySelector('#controlled-player'); document.querySelector('#play-btn').addEventListener('click', () => { player.play(); }); document.querySelector('#pause-btn').addEventListener('click', () => { player.pause(); }); document.querySelector('#stop-btn').addEventListener('click', () => { player.stop(); }); document.querySelector('#speed-up-btn').addEventListener('click', () => { player.speed = player.speed * 1.5; }); document.querySelector('#speed-down-btn').addEventListener('click', () => { player.speed = player.speed / 1.5; });
